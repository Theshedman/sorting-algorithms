\underline{\bf{Laufzeit}}

\begin{tabular}{|l|l|l|}

\underline{\bf{Pseudocode}}

\begin{lstlisting}
procedure versickere(a, i, m)
-----------------------------(d.h. versickere a[i] bis hoechstens a[m] im Heap a)
\end{lstlisting}

\begin{lstlisting}
Sortieren eines Heaps a von Position l bis r
--------------------------------------------begin  for i := r downto 2 do    begin (tausche a[l] mit a[i], versickere a[l]) 
      t:= a[i];      a[i] := a[l];      a[l] := t; 
      versickere(a, l, i - l)    end
end
\end{lstlisting}

\begin{lstlisting}
Erzeugen eines Heaps mit Laenge n
---------------------------------
begin  for i := [n/2] downto 1 do    versickere(a, i, n) 
end
\end{lstlisting}
