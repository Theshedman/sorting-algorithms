\underline{\bf{Pseudocode}}



\begin{lstlisting}
Selectionsort
-------------begin  for i := 1 to n - 1 do    begin
      min := i;
      for j := i + 1 to n do 
        if a[j].key < a[min].key
        then min := j;
      t:= a[min]; a[min]:= [i]; a[i] := t;    end
end
\end{lstlisting}
